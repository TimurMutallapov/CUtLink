CREATE TABLE IF NOT EXISTS public.lin
(
    id integer NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 2147483647 CACHE 1 ),
    mainlink character varying(100) COLLATE pg_catalog."default" NOT NULL,
    cutlink character varying(100) COLLATE pg_catalog."default" NOT NULL,
    CONSTRAINT lin_pkey PRIMARY KEY (id)
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.lin
    OWNER to postgres;
